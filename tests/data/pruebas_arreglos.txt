// CASO 1: Arreglo Correcto
// DESCRIPCION: Declaración y uso válido de arreglos.
int main() {
    int arreglo[10];
    for( int x=0;x<5;x++ ){
        if(x<10){
            arreglo[x]=x+1;
        }
    }
}
// SALIDA ESPERADA:
// [SUCCESS]

// CASO 2: Declaración Incorrecta (Falta tamaño)
// DESCRIPCION: int arreglo[]; no es válido.
int main() {
    int arreglo[];
    for( int x=0;x<5;x++ ){
        if(x<10){
            arreglo[x]=x+1;
        }
    }
}
// SALIDA ESPERADA:
// [ERROR] Error semántico: Declaración de arreglo inválida 'arreglo[]'. Se espera 'nombre[numero]'.

// CASO 3: Sintaxis Incorrecta (Orden invertido)
// DESCRIPCION: int [10] arreglo; no es válido.
int main() {
    int [10] arreglo;
    for( int x=0;x<5;x++ ){
        if(x<10){
            arreglo[x]=x+1;
        }
    }
}
// SALIDA ESPERADA:
// [ERROR] Error sintáctico
