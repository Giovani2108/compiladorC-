int main() {
    int arr[5];
    arr[0] = 64;
    arr[1] = 34;
    arr[2] = 25;
    arr[3] = 12;
    arr[4] = 22;

    int i;
    int j;
    int temp;
    
    cout << 9999; // Marker start

    for (i = 0; i < 5; i++) {
        for (j = 0; j < 4; j++) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    cout << 8888; // Marker sorted
    
    for (i = 0; i < 5; i++) {
        cout << arr[i];
    }
}
